moda <- function(x){
    ux <- unique(x)
    ux[which.max(tabulate(match(x, ux)))]
}

uploadBox <- function(id){
    ns <- NS(id)
    box(
        title = "Carga de datos", width = 12, status = "primary", solidHeader = TRUE,
        fileInput(ns("file"), "Selecciona archivo", accept = c(".csv", ".txt", ".xlsx", ".xls", ".sav")),
        selectInput(ns("sep"), "Separador columnas", c("Coma"=",","Punto y coma"=";","Tab"="    ")),
        selectInput(ns("dec"), "Separador decimal", c("Punto"=".","Coma"=",")),
        checkboxInput(ns("header"), "Cabecera", TRUE),
        br(),
        h4("Vista previa (3 primeras filas)"),
        tableOutput(ns("preview"))
    )
}

uploadServer <- function(id){
    moduleServer(id, function(input, output, session){
        
        observe({
            req(input$file)
            ext <- tolower(tools::file_ext(input$file$name))
            disable_opts <- ext %in% c("xlsx", "xls", "sav")
            
            shinyjs::toggleState("sep", condition = !disable_opts)
            shinyjs::toggleState("dec", condition = !disable_opts)
            shinyjs::toggleState("header", condition = !disable_opts)
        })
        data_react <- reactive({
            req(input$file)
            ext <- tolower(tools::file_ext(input$file$name))
            
            df <- switch(ext,
                         "sav" = as.data.frame(read_sav(input$file$datapath)),
                         "xlsx" = as.data.frame(read_excel(input$file$datapath)),
                         "xls" = as.data.frame(read_excel(input$file$datapath)),
                         read.table(input$file$datapath,
                                    sep = input$sep,
                                    dec = input$dec,
                                    header = input$header,
                                    stringsAsFactors = FALSE,
                                    check.names = FALSE)
            )
            
            # validate(
            #     need(ncol(df) > 1 || ext %in% c("xlsx","xls","sav"),
            #          "La estructura de los datos no parece correcta. Revisa separadores y cabecera.")
            # )
            
            validate(
                need(ncol(df) > 1 || ext %in% c("xlsx","xls","sav"),
                     "")
            )
            
            as.data.frame(df)
        })
        
        output$preview <- renderTable({
            req(data_react())
            head(data_react(), 3)
        })
        
        return(data_react)
    })
}

# ========================= SERVER
# =========================
server <- function(input, output, session){
    
    data_desc <- uploadServer("up_desc")
    data_reg  <- uploadServer("up_reg")
    data_ci   <- uploadServer("up_ci")
    data_test <- uploadServer("up_test")
    
    # ---------- DESCRIPTIVAS ----------
    output$vars_desc_ui <- renderUI({
        req(data_desc())
        req(data_desc())
        nums <- names(data_desc())[sapply(data_desc(), is.numeric)]
        selectInput("vars_desc", "Variables", nums, multiple = TRUE)
    })
    
    observe({
        if(isTRUE(input$show_hist)){
            shinyjs::enable("normal")
        } else {
            shinyjs::disable("normal")
            updateCheckboxInput(session, "normal", value = FALSE)
        }
    })
    
    output$tabla_desc <- renderTable({
        req(input$vars_desc)
        df <- data_desc()
        res <- list()
        percentiles <- if(nchar(input$percent)>0) as.numeric(strsplit(input$percent,",")[[1]]) else NULL
        
        for(v in input$vars_desc){
            x <- na.omit(df[[v]])
            tmp <- c()
            if(input$mean) tmp['Media'] <- mean(x)
            if(input$median) tmp['Mediana'] <- median(x)
            if(input$mode) tmp['Moda'] <- moda(x)
            if(input$quart){
                q <- quantile(x, c(.25,.5,.75))
                tmp['Q1'] <- q[1]; tmp['Q2'] <- q[2]; tmp['Q3'] <- q[3]
            }
            if(!is.null(percentiles)){
                q <- quantile(x, percentiles/100)
                for(i in seq_along(q)) tmp[paste0('P', percentiles[i])] <- q[i]
            }
            if(input$min) tmp['Mínimo'] <- min(x)
            if(input$max) tmp['Máximo'] <- max(x)
            if(input$range) tmp['Rango'] <- max(x) - min(x)
            if(input$iqr) tmp['Rango intercuartílico'] <- IQR(x)
            if(input$var) tmp['Varianza'] <- var(x)
            if(input$sd) tmp['SD'] <- sd(x)
            if(input$cv) tmp['Coef. variación'] <- sd(x)/mean(x)
            if(input$skew) tmp['Asimetría'] <- skewness(x)
            if(input$kurt) tmp['Curtosis'] <- kurtosis(x)
            res[[v]] <- tmp
        }
        as.data.frame(res)
    }, rownames = TRUE)
    
    output$plots_desc <- renderUI({
        req(input$vars_desc)
        
        graficos <- list()
        
        for(v in input$vars_desc){
            
            if(input$show_hist){
                graficos <- append(graficos, list(
                    plotOutput(paste0("hist_", v), height = 250)
                ))
            }
            
            if(input$show_box){
                graficos <- append(graficos, list(
                    plotOutput(paste0("box_", v), height = 250)
                ))
            }
        }
        
        graficos
    })
    
    observe({
        req(input$vars_desc)
        
        df <- data_desc()
        
        for(v in input$vars_desc){
            
            local({
                varname <- v
                
                output[[paste0("hist_", varname)]] <- renderPlot({
                    
                    req(input$show_hist)
                    
                    x <- na.omit(df[[varname]])
                    
                    hist(
                        x,
                        probability = TRUE,
                        main = paste("Histograma", varname),
                        col = "gray80"
                    )
                    
                    if(input$normal){
                        curve(
                            dnorm(x, mean(x), sd(x)),
                            add = TRUE,
                            col = "red",
                            lwd = 2
                        )
                    }
                })
                
                output[[paste0("box_", varname)]] <- renderPlot({
                    
                    req(input$show_box)
                    
                    x <- na.omit(df[[varname]])
                    
                    boxplot(
                        x,
                        horizontal = FALSE,
                        main = paste("Boxplot", varname)
                    )
                })
            })
        }
    })
    
    # ---------- REGRESIÓN ----------
    output$dep_ui <- renderUI({
        req(data_reg())
        
        nums <- names(data_reg())[sapply(data_reg(), is.numeric)]
        
        selectInput(
            "dep_var",
            "Dependiente",
            choices = c("Selecciona variable" = "", nums),
            selected = ""
        )
    })
    
    output$indep_ui <- renderUI({
        req(data_reg(), input$dep_var)
        
        nums <- names(data_reg())[sapply(data_reg(), is.numeric)]
        nums <- nums[nums != input$dep_var]
        
        selectInput(
            "indep_vars",
            "Independientes",
            nums,
            multiple = TRUE
        )
    })
    
    observe({
        if(length(input$indep_vars) == 1){
            shinyjs::enable("show_scatter")
        } else {
            shinyjs::disable("show_scatter")
            updateCheckboxInput(session, "show_scatter", value = FALSE)
        }
    })
    
    output$reg_table <- renderTable({
        req(input$dep_var != "", input$indep_vars)
        
        fit <- lm(
            as.formula(
                paste(input$dep_var, "~", paste(input$indep_vars, collapse = "+"))
            ),
            data = data_reg()
        )
        
        coefs <- as.data.frame(summary(fit)$coefficients)
        
        coefs$Coeficiente <- rownames(coefs)
        rownames(coefs) <- NULL
        
        coefs <- coefs[, c(
            "Coeficiente",
            "Estimate",
            "Std. Error",
            "t value",
            "Pr(>|t|)"
        )]
        
        names(coefs) <- c(
            "Coeficiente",
            "Estimación",
            "Error Std",
            "t",
            "p_value"
        )
        
        coefs[, -1] <- round(coefs[, -1], 5)
        
        coefs
    }, rownames = FALSE)
    
    output$corr_table <- renderTable({
        req(input$corr_matrix, input$dep_var, input$indep_vars)
        
        vars <- c(input$dep_var, input$indep_vars)
        
        round(
            cor(data_reg()[vars], use = "pairwise.complete.obs"),
            5
        )
        
    }, rownames = TRUE)
    
    output$cov_table <- renderTable({
        req(input$cov_matrix, input$dep_var != "", input$indep_vars)
        
        vars <- c(input$dep_var, input$indep_vars)
        
        round(
            cov(data_reg()[vars], use = "pairwise.complete.obs"),
            5
        )
        
    }, rownames = TRUE)
    
    output$r2_table <- renderTable({
        req(input$r2_show, input$dep_var != "", input$indep_vars)
        
        fit <- lm(
            as.formula(
                paste(input$dep_var, "~", paste(input$indep_vars, collapse = "+"))
            ),
            data = data_reg()
        )
        
        data.frame(
            Medida = c("R²", "R² ajustado"),
            Valor = round(
                c(summary(fit)$r.squared,
                  summary(fit)$adj.r.squared),
                5
            )
        )
        
    }, rownames = FALSE)
    
    output$corr_test_table <- renderTable({
        req(input$corr_test, input$dep_var, input$indep_vars)
        
        vars <- c(input$dep_var, input$indep_vars)
        
        if(length(vars) < 2) return(NULL)
        
        cmb <- combn(vars, 2, simplify = FALSE)
        
        do.call(rbind, lapply(cmb, function(p){
            
            ct <- cor.test(
                data_reg()[[p[1]]],
                data_reg()[[p[2]]]
            )
            
            data.frame(
                Variable_1 = p[1],
                Variable_2 = p[2],
                Correlacion = round(ct$estimate, 5),
                p_value = round(ct$p.value, 5)
            )
        }))
        
    }, rownames = FALSE)
    
    output$shapiro_table <- renderTable({
        req(input$do_shapiro, input$dep_var)
        
        vars <- c(input$dep_var, input$indep_vars)
        
        do.call(rbind, lapply(vars, function(v){
            
            x <- na.omit(data_reg()[[v]])
            
            if(length(x) < 3 || length(x) > 5000){
                return(data.frame(
                    Variable = v,
                    W = NA,
                    p_value = NA
                ))
            }
            
            sh <- shapiro.test(x)
            
            data.frame(
                Variable = v,
                W = round(sh$statistic, 5),
                p_value = round(sh$p.value, 5)
            )
        }))
        
    }, rownames = FALSE)
    
    output$qq_ui <- renderUI({
        req(input$do_qq, input$dep_var)
        
        vars <- c(input$dep_var, input$indep_vars)
        
        lapply(vars, function(v){
            plotOutput(paste0("qq_", v), height = 250)
        })
    })
    
    observe({
        req(input$do_qq, input$dep_var)
        
        vars <- c(input$dep_var, input$indep_vars)
        
        for(v in vars){
            local({
                varname <- v
                
                output[[paste0("qq_", varname)]] <- renderPlot({
                    x <- na.omit(data_reg()[[varname]])
                    
                    qqnorm(x, main = paste("Q-Q Plot:", varname))
                    qqline(x, col = "red", lwd = 2)
                })
            })
        }
    })
    
    output$scatter_plot <- renderPlot({
        req(input$show_scatter)
        req(input$dep_var)
        
        if(length(input$indep_vars) != 1) return(NULL)
        
        ggplot(
            data_reg(),
            aes_string(
                x = input$indep_vars[1],
                y = input$dep_var
            )
        ) +
            geom_point() +
            geom_smooth(method = "lm", se = FALSE, color = "red") +
            theme_minimal()
    })
    
    # ---------- IC ----------
    output$vars_ci_ui <- renderUI({
        req(data_ci())
        nums <- names(data_ci())[sapply(data_ci(), is.numeric)]
        selectInput("vars_ci", "Variables", nums, multiple = TRUE)
    })
    
    output$group_var_ui <- renderUI({
        req(data_ci())
        
        df <- data_ci()
        
        # variables NO numéricas → factores
        facs <- names(df)[!sapply(df, is.numeric)]
        
        if(length(facs) == 0) return(NULL)
        
        selectInput(
            "group_var",
            "Variable de agrupación",
            choices = c("Ninguna" = "", facs),
            selected = ""
        )
    })
    
    observe({
        
        # número de variables numéricas seleccionadas
        n_vars <- length(input$vars_ci)
        
        # si no hay agrupación
        no_group <- is.null(input$group_var) || input$group_var == ""
        
        # IC diferencia medias
        if(n_vars < 1 || no_group){
            shinyjs::disable("ci_diff")
            updateCheckboxInput(session, "ci_diff", value = FALSE)
        } else {
            shinyjs::enable("ci_diff")
        }
        
        # IC cociente varianzas
        if(n_vars < 1 || no_group){
            shinyjs::disable("ci_ratio_var")
            updateCheckboxInput(session, "ci_ratio_var", value = FALSE)
        } else {
            shinyjs::enable("ci_ratio_var")
        }
    })
    
    output$ci_meanvar_table <- renderTable({
        req(input$vars_ci)
        
        conf <- ifelse(is.null(input$conf_level), 0.95, input$conf_level/100)
        alpha <- 1 - conf
        
        res <- list()
        
        for(v in input$vars_ci){
            x <- na.omit(data_ci()[[v]])
            n <- length(x)
            m <- mean(x)
            s2 <- var(x)
            
            fila <- data.frame(Variable = v)
            
            if(input$ci_mean){
                tval <- qt(1 - alpha/2, n - 1)
                li <- m - tval * sqrt(s2/n)
                ls <- m + tval * sqrt(s2/n)
                
                fila$Media <- round(m,5)
                fila$LI_Media <- round(li,5)
                fila$LS_Media <- round(ls,5)
            }
            
            if(input$ci_var){
                chi1 <- qchisq(alpha/2, n-1)
                chi2 <- qchisq(1 - alpha/2, n-1)
                
                li <- (n-1)*s2/chi2
                ls <- (n-1)*s2/chi1
                
                fila$Varianza <- round(s2,5)
                fila$LI_Var <- round(li,5)
                fila$LS_Var <- round(ls,5)
            }
            
            res[[v]] <- fila
        }
        
        do.call(rbind, res)
    }, rownames = FALSE)
    
    
    output$ci_diff_table <- renderTable({
        req(input$ci_diff, input$vars_ci, input$group_var)
        
        df <- data_ci()
        g <- as.factor(df[[input$group_var]])
        
        if(length(unique(g)) != 2) return(NULL)
        
        niveles <- levels(g)
        conf <- input$conf_level/100
        alpha <- 1 - conf
        
        res <- list()
        
        for(v in input$vars_ci){
            
            x1 <- na.omit(df[g == niveles[1], v])
            x2 <- na.omit(df[g == niveles[2], v])
            
            n1 <- length(x1); n2 <- length(x2)
            m1 <- mean(x1); m2 <- mean(x2)
            s1 <- var(x1); s2 <- var(x2)
            
            sp <- ((n1-1)*s1 + (n2-1)*s2)/(n1+n2-2)
            se <- sqrt(sp*(1/n1 + 1/n2))
            
            tval <- qt(1 - alpha/2, n1+n2-2)
            
            diff <- m1 - m2
            
            li <- diff - tval * se
            ls <- diff + tval * se
            
            res[[v]] <- data.frame(
                Variable = v,
                Media_Grupo1 = round(m1,5),
                Media_Grupo2 = round(m2,5),
                Diferencia = round(diff,5),
                LI = round(li,5),
                LS = round(ls,5)
            )
        }
        
        do.call(rbind, res)
    }, rownames = FALSE)
    
    
    output$ci_ratio_table <- renderTable({
        req(input$ci_ratio_var, input$vars_ci, input$group_var)
        
        df <- data_ci()
        g <- as.factor(df[[input$group_var]])
        
        if(length(unique(g)) != 2) return(NULL)
        
        niveles <- levels(g)
        conf <- input$conf_level/100
        alpha <- 1 - conf
        
        res <- list()
        
        for(v in input$vars_ci){
            
            x1 <- na.omit(df[g == niveles[1], v])
            x2 <- na.omit(df[g == niveles[2], v])
            
            n1 <- length(x1); n2 <- length(x2)
            s1 <- var(x1); s2 <- var(x2)
            
            ratio <- s1/s2
            
            f1 <- qf(alpha/2, n1-1, n2-1)
            f2 <- qf(1 - alpha/2, n1-1, n2-1)
            
            li <- ratio / f2
            ls <- ratio / f1
            
            res[[v]] <- data.frame(
                Variable = v,
                Var_Grupo1 = round(s1,5),
                Var_Grupo2 = round(s2,5),
                Cociente = round(ratio,5),
                LI = round(li,5),
                LS = round(ls,5)
            )
        }
        
        do.call(rbind, res)
    }, rownames = FALSE)
