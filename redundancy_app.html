# Servidor
server <- function(input, output) {
    observeEvent(input$calc_button, {
        if (trimws(input$p_values) == "") {
            output$error_message <- renderText("Error: Please enter reliability components in the field.")
            return()
        }
        
        p <- as.numeric(unlist(strsplit(input$p_values, ",")))
        R_limit <- input$R_limit
        
        if (all(!is.na(p)) && all(p >= 0 & p <= 1) && R_limit >= 0 && R_limit <= 1) {
            result <- Redundance(p, R_limit)
            
            output$R_ini_output <- renderText({ paste("Initial reliability:", format(result$R.ini, digits = 5, nsmall = 5)) })
            output$R_optim_output <- renderText({ paste("Optimized reliability:", format(result$R.optim[length(result$R.optim)], digits = 5, nsmall = 5)) })
            output$n_replicas_output <- renderText({ paste("Number of replicas:", (result$n.replicas) - 1) })
            
            output$table_ui <- renderUI({
                table <- renderTable({
                    mat <- as.data.frame(result$mat.optim)
                    colnames(mat) <- paste0("Component ", 1:ncol(mat))
                    mat$`System reliability` <- format(result$R.optim, digits = 5, nsmall = 5)
                    rownames(mat) <- c("Initial reliabilities", paste0("Replica ", 1:(nrow(mat) - 1)))
                    mat
                }, rownames = TRUE, colnames = TRUE, align = "c")
                tags$div(style = "display: flex; justify-content: center; text-align: center;", table)
            })
            
            output$reliability_plot <- renderPlot({
                plot(1:length(result$R.optim), result$R.optim, type = "o", xlab = "Number of replicas",
                     ylab = "System reliability", main = "Evolution of reliability with redundancy", col = "blue", pch = 19)
                abline(h = R_limit, col = "red", lty = 2)
                legend("bottomright", legend = c("Reliability", "Limit"), col = c("blue", "red"), lty = c(1, 2), pch = c(19, NA))
            })
            
            output$error_message <- renderText("")
        } else {
            output$error_message <- renderText("Error: Enter valid p values in [0, 1] and ensure the reliability limit is within [0, 1].")
        }
    })
}

# Funciones auxiliares
s.p <- function(mat.p) {
    n.col <- dim(mat.p)[2]
    R.col <- double(n.col)
    for (i in 1:n.col) {
        R.col[i] <- (1 - prod(1 - mat.p[, i]))
    }
    R <- prod(R.col)
    R
}

Redundance <- function(p, R.limit) {
    n <- length(p)
    mat.optim <- matrix(p, 1, n)
    
    R.ini <- s.p(mat.optim)
    R.optim <- c(R.ini)
    k <- 1
    
    while (R.optim[k] < R.limit) {
        R.aux <- double(n)
        k <- k + 1
        for (i in 1:n) {
            mat.aux <- rbind(mat.optim, rep(0, n))
            mat.aux[k, i] <- p[i]
            R.aux[i] <- s.p(mat.aux)
        }
        i.optim <- which.max(R.aux)
        R.optim <- c(R.optim, R.aux[i.optim])
        p.optim <- double(n); p.optim[i.optim] <- p[i.optim]
        mat.optim <- rbind(mat.optim, p.optim)
    }
    
    return(list(R.ini = R.ini, R.optim = R.optim, mat.optim = mat.optim, n.replicas = k))
}
